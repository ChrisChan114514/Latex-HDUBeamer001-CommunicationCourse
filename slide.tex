\begin{frame}[t]{一、从多址接入看CDMA}

\begin{frame}[t]{二、CDMA工作原理：扩频与解扩}
    \small
    \begin{block}{核心思想}
        CDMA不是把用户分到不同时间片，而是给每个用户分配不同码字。发送端扩频，接收端用同一码字解扩。
    \end{block}

    \vspace{-0.6em}
    \[
        s_i(t)=b_i(t)c_i(t), \qquad
        \int c_i(t)c_j(t)\,dt \approx 0\quad(i\neq j)
    \]

    \vspace{-0.8em}
    \centering

        \node[flowbox, text width=2.1cm] (data) at (0,0.78) {信息比特\\$b_i(t)$};
        \node[flowbox, text width=2.1cm] (code) at (0,-0.78) {扩频码\\$c_i(t)$};
        \node[orangebox, text width=2.0cm] (spread) at (3.0,0) {相乘\\扩频};

        \node[flowbox, text width=2.1cm] (out) at (11.5,0) {恢复信息\\$\hat b_i(t)$};

\end{frame}

\begin{frame}[t]{二、CDMA工作原理：接收端如何分离用户}
    \small
    \begin{block}{接收端看到的是叠加信号}
        空中同时存在多个用户信号。接收机只关心目标用户码字，通过相关运算把目标用户“提出来”。
    \end{block}

\begin{frame}[t]{三、CDMA相对TDMA的优势：移动与抗干扰}
    \small
    \begin{columns}[T,totalwidth=\textwidth]
        \column{0.33\textwidth}
        \begin{block}{频率复用}
            相邻小区可使用相同频带，邻区信号主要表现为附加干扰，而不是必须完全避开。
        \end{block}

        \column{0.33\textwidth}
        \begin{block}{抗多径}
            扩频信号配合RAKE接收，可把可分辨多径能量合并，而不是只把多径当成失真。
        \end{block}

        \column{0.33\textwidth}
        \begin{block}{软切换}
            小区边缘可同时保持多个基站链路，先连新链路再释放旧链路，降低掉话概率。
        \end{block}
    \end{columns}

\begin{frame}[t]{七、N-CDMA反向链路：基本特点}
    \small
    \begin{columns}[T,totalwidth=\textwidth]
        \column{0.42\textwidth}
        \begin{block}{反向链路为什么更难？}
            多个移动台从不同位置同时向同一基站发送。它们彼此不同步，而且离基站远近不同。
        \end{block}

        \begin{alertblock}{关键约束}
            系统必须通过功率控制，让不同移动台到达基站的功率尽量接近，否则近处用户会淹没远处用户。
        \end{alertblock}

\section{N-CDMA关键技术}

\begin{frame}[t]{八、N-CDMA关键技术总览}
    \small
    \begin{columns}[T,totalwidth=\textwidth]
        \column{0.50\textwidth}
        \begin{block}{让用户能同频共存}
            \keypoint{扩频与解扩}：发送端用高速码扩频，接收端用同码相关恢复目标用户。
        \end{block}

        \begin{block}{让干扰保持可控}
            \keypoint{功率控制}：让不同移动台到达基站的功率接近，压制远近效应。
        \end{block}

        \column{0.50\textwidth}
        \begin{block}{让多径变成可利用能量}
            \keypoint{RAKE接收}：分别捕获可分辨多径，再按质量加权合并。
        \end{block}

        \begin{block}{让移动过程不中断}
            \keypoint{软切换}：小区边缘先建立新链路，再释放旧链路，降低掉话概率。
        \end{block}
    \end{columns}
\end{frame}

\begin{frame}[t]{八、N-CDMA关键技术如何配合}
    \small
    \begin{columns}[T,totalwidth=\textwidth]
        \column{0.46\textwidth}
        \begin{block}{底层可靠性}
            \keypoint{信道编码与交织}给数据加冗余，并把突发错误打散，便于接收端纠错。
        \end{block}

        \begin{block}{系统干扰管理}
            \keypoint{语音激活}在用户不说话时降低发送，减少总干扰，提高系统容量。
        \end{block}

        \begin{block}{接入与小区识别}
            \keypoint{同步与PN规划}帮助终端找到小区，也避免相邻小区码字混淆。
        \end{block}

\begin{frame}[t]{九、功率控制：解决远近效应}
    \scriptsize
    \begin{columns}[T,totalwidth=\textwidth]
        \column{0.48\textwidth}
        \begin{block}{远近效应}
            如果近处和远处移动台用相同功率发射，近处信号到达基站会强得多，强用户会成为弱用户的严重干扰。
        \end{block}

        \vspace{0.1em}
        \begin{itemize}
            \item 开环控制：移动台根据接收导频强弱估计初始发射功率。
            \item 闭环控制：基站快速命令移动台升功率或降功率。
            \item 目标不是“功率越大越好”，而是\keypoint{刚好够用且不过度干扰别人}。
        \end{itemize}

\end{frame}

\section{总结}

\begin{frame}[t]{总结（一）：CDMA与N-CDMA}
    \small
    \begin{enumerate}
        \item CDMA的本质是用码字而不是时间或频率来区分用户。
        \item 与TDMA相比，CDMA更强调软容量、抗多径、软切换和频率复用。
        \item N-CDMA系统由移动台、基站、基站控制器、交换中心、数据库和网管系统共同构成。
        \item Walsh码、短PN码和长PN码分别承担信道区分、小区识别、用户识别等任务。
    \end{enumerate}
\end{frame}

\begin{frame}[t]{总结（二）：信道、关键技术与3G制式}
    \small
    \begin{enumerate}
        \setcounter{enumi}{4}
        \item 正向链路重点是导频、同步、寻呼和业务信道；反向链路重点是接入和业务信道。
        \item 功率控制、RAKE接收、软切换、编码交织、语音激活和同步规划共同支撑系统容量。
        \item WCDMA、CDMA2000、TD-SCDMA都是3G代表制式，但频谱组织、同步要求和演进路径不同。
    \end{enumerate}
\end{frame}
